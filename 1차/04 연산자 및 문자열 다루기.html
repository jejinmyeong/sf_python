#문자열 위치 찾기(find 또는 index)
#rfind 또는 rindex : 오른쪽에서부터 문자열 위치 찾기
s='apple pineapple'
print(s.find('pine'))
print(s.find('a',5)) # s[5:] 슬라이싱처럼 찾음
print(s.find('ap',5,-1)) #s[5:-1] 슬라이싱
print(s.find('5')) #찾는 값이 없으면 -1을 반환
print(s.index('5')) #index함수는 find함수와 동일하지만 찾는 값이 없으면 에러 발생
